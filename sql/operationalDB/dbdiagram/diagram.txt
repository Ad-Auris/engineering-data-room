Table users[headercolor: #d35400] {
    id VARCHAR(55) //[pk] // primary key PRIMARY KEY NOT NULL
    username VARCHAR(255) // NOT NULL
    googleId VARCHAR(255)
    tier INT
    organisation_id INTEGER //REFERENCES organisations(id)
    num_ttsp INT //NOT NULL DEFAULT 0
    date_created TIMESTAMP //NOT NULL DEFAULT CURRENT_TIMESTAMP
    email VARCHAR(255) //NOTNULL
    country VARCHAR(255)
    audience_country VARCHAR(255)
    topic VARCHAR(255)
    demographic VARCHAR(255)
    desire VARCHAR(255)
    role VARCHAR(55)
    last_signed_in TIMESTAMP //NOT NULL DEFAULT CURRENT_TIMESTAMP

}

Table organisations[headercolor: #27ae60] {
    id SERIAL[pk] // primary key
    name VARCHAR(255) UNIQUE
    route_identifier VARCHAR(255)
    admin_id VARCHAR(255) //REFERENCES users(id)
    property_id VARCHAR(255)
    default_view_id INT
    filter_view_id INT
    date TIMESTAMP // NOT NULL DEFAULT CURRENT_TIMESTAMP
    join_code VARCHAR(255)
}

    title BOOLEAN // DEFAULT false
    image TEXT // DEFAULT ''
    label TEXT
    font_family TEXT
    font_colour TEXT
    font_size TEXT
    font_weight TEXT
    logo_colour TEXT
}


Table rss_id[headercolor: #c0392b] {

Table tts_packages[headercolor: #16a085] {

    user_id VARCHAR(255) //REFERENCES users(id),

    title VARCHAR(255)// NOT NULL DEFAULT '',
    route_identifier VARCHAR(255)
    file VARCHAR(255) //NOT NULL DEFAULT '',
    original_text TEXT //NOT NULL DEFAULT '',
    modified_text VARCHAR(255) //NOT NULL DEFAULT '',

    comments TEXT
    sentiment_analysis TEXT
    emotion_analysis TEXT
    keywords_analysis TEXT
    widget_image TEXT
    footer_title TEXT
    narration_method TEXT //DEFAULT 'Instant',
    article_url TEXT
    title_voice TEXT
    music TEXT //NOT NULL DEFAULT '',
    ct_link TEXT
    experiment INTEGER //REFERENCES experiment(id)
}


Table audio[headercolor: #16a085] {

Table removal_config[headercolor: #8e44ad] {
    id SERIAL //PRIMARY KEY NOT NULL,
    narration_setting_id INTEGER //REFERENCES narration_settings_id(id),
    removal_type TEXT // E.G:  "Remove First Instance by HTML Tag"
    start_html_statement TEXT // E.G:  "<strong>"
    end_html_tag TEXT // E.G:  "</strong>"
    settings_name TEXT // E.G: Remove Double Author Blurb
}

Table convert_html_statement[headercolor: #8e44ad] {
    id SERIAL //PRIMARY KEY NOT NULL,
    narration_setting_id INTEGER //REFERENCES narration_settings_id(id),
    removal_type TEXT // E.G:  "Remove First Instance by HTML Tag"
    start_html_statement TEXT // E.G:  "<strong><p>"
    end_html_statement TEXT // E.G:  "</strong></p>"
    converted_start_html_statement TEXT // E.G:  "<em>"
    converted_end_html_statement TEXT // E.G:  "</em>"
    settings_name TEXT // E.G: Convert strong + p tags into italics tags
}

Table replace_html_statements[headercolor: #8e44ad] {
    id SERIAL //PRIMARY KEY NOT NULL,
    narration_setting_id INTEGER //REFERENCES narration_settings_id(id),
    removal_type TEXT // E.G:  "Remove First Instance by HTML Tag"
    start_html_statement TEXT // E.G:  "<figure>"
    end_html_statement TEXT // E.G:  "</figure>"
    replaced_statement TEXT // E.G:  "<em>"
    converted_end_html_statement TEXT // E.G:  "<p>Look on screen for graphics</P"
    settings_name TEXT // E.G: Tell listener to look on screen for media
}

Table omit_slice {
    id SERIAL
    marker TEXT
    remove_below BOOLEAN
    narration_setting_id INTEGER //FK
}

Table replacement {
    id SERIAL
    name TEXT
    regex TEXT
    marker TEXT
    replacer TEXT
    intent replacement_intent // all, first
    narration_setting_id INTEGER //FK
}
